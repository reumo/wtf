const [country, setCountry] = useState("EN"); const [car, setCar] = useState("Ferrari");

useEffect(() => {
  const urlParams =
      new URLSearchParams(window.location.search);
  if (urlParams.has("country")) {     
    setCountry(urlParams.get("country"));
  }
  if (urlParams.has("car")) {
    setCar(urlParams.get("car"));
  }}, []);

Explanation:
  • * window.location.search => returns the query portion of a URL including the Question mark: ?country=ES&car=lamborghini.
  • * The URLSearchParams interface defines utility methods to work with the query string of a URL.
  • * urlParams.has('country') verifies if the queryParam "country" exists.
  • * urlParams.get('country') returns queryParam "country" value.
